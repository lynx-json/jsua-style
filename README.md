JSUA Select
=================================================

```js

function select(selector) {
  var s = other.select(selector);
  s.
}

finishing.register([
  r => select(r)
    .select(":first-child")
    .filter(el => el.dataset.uxVisibility === "hidden")
    .map(el => el.children)
    .each(el => el.dataset.child = true)
    .map(el => el.parentElement)
    .each(el => el.dataset.parent = true)
    .each(material.elevation(3));
]);


```
